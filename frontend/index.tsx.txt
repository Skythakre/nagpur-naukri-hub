import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, Alert } from 'react-native';
import { auth } from '../lib/firebase';
import { PhoneAuthProvider, signInWithCredential } from 'firebase/auth';

export default function LoginScreen() {
  const [phoneNumber, setPhoneNumber] = useState('');
  const [otp, setOtp] = useState('');
  const [verificationId, setVerificationId] = useState('');
  const [step, setStep] = useState(1); // 1: Phone, 2: OTP

  // OTP bhejne ka function
  const sendOtp = async () => {
    if (phoneNumber.length < 10) {
      Alert.alert("Error", "Valid Nagpur mobile number dalein");
      return;
    }
    try {
      // Yahan Firebase phone auth trigger hoga
      // Note: Real device par ye SMS bhejega
      console.log("Sending OTP to:", phoneNumber);
      setStep(2); 
    } catch (error) {
      Alert.alert("Auth Error", error.message);
    }
  };

  // OTP verify karne ka function
  const verifyOtp = async () => {
    try {
      // Yahan OTP verify hoga
      Alert.alert("Success", "Nagpur Naukri Hub mein aapka swagat hai!");
      // Iske baad user Home screen par chala jayega
    } catch (error) {
      Alert.alert("Invalid OTP", "Kripya sahi code dalein");
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Nagpur Naukri Hub üçä</Text>
      
      {step === 1 ? (
        <View>
          <Text style={styles.subtitle}>Apna Mobile Number Dalein</Text>
          <TextInput
            style={styles.input}
            placeholder="+91 00000-00000"
            keyboardType="phone-pad"
            value={phoneNumber}
            onChangeText={setPhoneNumber}
          />
          <TouchableOpacity style={styles.button} onPress={sendOtp}>
            <Text style={styles.buttonText}>OTP Mangwayein</Text>
          </TouchableOpacity>
        </View>
      ) : (
        <View>
          <Text style={styles.subtitle}>6-Digit OTP Dalein</Text>
          <TextInput
            style={styles.input}
            placeholder="123456"
            keyboardType="number-pad"
            value={otp}
            onChangeText={setOtp}
          />
          <TouchableOpacity style={styles.button} onPress={verifyOtp}>
            <Text style={styles.buttonText}>Verify & Login</Text>
          </TouchableOpacity>
          <TouchableOpacity onPress={() => setStep(1)}>
            <Text style={styles.link}>Number badlein?</Text>
          </TouchableOpacity>
        </View>
      )}
    </View>
  );
}
